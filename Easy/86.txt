//the question is asking us to replace the vowels in the text with their mapping
//A → V , U → V , E → X , O → X , I → I
//can be done in two ways->

//using regex
const text = readline();
const result = text.replace(/[AEIOU]/g, (ch) => {
        if (ch === 'A' || ch === 'U') return 'V';
        if (ch === 'E' || ch === 'O') return 'X';
        return 'I'; 
});
console.log(result);

//or using loop and mapping
const text = readline();
const map = {
        'A': 'V',
        'U': 'V', 
        'E': 'X',
        'O': 'X'
}
let result = "";
for(let ch of text){
        result += map[ch] ?? ch; 
}
console.log(result);
