//the first line of input is the number of strings to input
const n = parseInt(readline());

//the question is asking us to print the number of digits in each string in separate lines
for(let i=1;i<=n;i++){
        const s = readline();
        let count = 0;
        for(let ch of s){
                if(ch>='0' && ch<='9') count++;
        }
        console.log(count);
}
